#include <iostream>
#include <fstream>
#define Length 10
using namespace std;

int main()
{
    int count = 0;
    float Arr[Length];

    char file_name[30];
    cout << "File name: "; 
    cin >> file_name;

    ifstream File(file_name);
    if (!File) cout << "file is not open.\n";
    else {
        while (!File.eof()) {
            File >> Arr[count];
            count++;
                
        }
        cout << "file contents: ";
        for (int i = 0; i < count - 1; i++) {
            cout << Arr[i] << " ";
        }
        cout << "\n";
        File.close();
    }

    float sum = 0;
    for (int i = 0; i < count - 1; i++) {
        sum += Arr[i];
    }
    cout << "Sum of number is: " << sum << "\n";
}
